About "this":

check this: Scope in javascript

Digest:

  • Calling an object's method: like C++, java:
    <script type="text/javascript"> 
      var deep_thought = { 
       the_answer: 42, 
       ask_question: function () { 
        return this.the_answer; 
       } 
      }; 
      var the_meaning = deep_thought.ask_question(); 
    </script>
  • Function constructor:
    <script type="text/javascript"> 
      function BigComputer(answer) { 
       this.the_answer = answer; 
       this.ask_question = function () { 
        return this.the_answer; 
       } 
      }   
      var deep_thought = new BigComputer(42); 
      var the_meaning = deep_thought.ask_question(); 
    </script>
  • Normal function call:
    <script type="text/javascript"> 
      function test_this() { 
       return this; 
      }
      var i_wonder_what_this_is = test_this();//this points to window
    </script>
  • Note the following case:
    <script type="text/javascript"> 
     function BigComputer(answer) { 
      this.the_answer = answer; 
      this.ask_question = function () { 
       alert(this.the_answer); 
      } 
     } 
      
     function addhandler() { 
      var deep_thought = new BigComputer(42), 
       the_button = document.getElementById('thebutton'); 
      
      the_button.onclick = deep_thought.ask_question; //will display undefined. "this" points to the_button
     } 
      
     window.onload = addhandler;
    </script>
